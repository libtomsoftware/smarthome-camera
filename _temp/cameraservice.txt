#!/bin/sh
#/etc/init.d/cameraservice
export PATH=$PATH:/usr/local/bin
export NODE_PATH=$NODE_PATH:/usr/local/lib/node_modules

case "$1" in
start)
exec forever --sourceDir=~/smarthome/camera -p ~/smarthome/camera index.js  #scriptarguments
;;
stop)
exec forever stop --sourceDir=~/smarthome/camera index.js
;;
*)
echo "Usage: /etc/init.d/cameraservice {start|stop}"
exit 1
;;
esac
exit 0
